package annotations;
import java.lang.annotation.*;
@Retention(RetentionPolicy.RUNTIME)
@Target(ElementType.FIELD)
public @interface FixedField {
    int length();
    boolean required() default true;
    String allowedValues() default "";
    String format() default "";
}


package model;

import annotations.FixedField;
public class CustomerDetails {
    @FixedField(length = 5,format = "\\d{5}")
    public String customerId;
    @FixedField(length = 20 )
    public String customerName;
    @FixedField(length = 6,allowedValues = "Male,Female,Others")
    public String gender;
    @FixedField(length = 10,format = "\\d{2}-\\d{2}-\\d{4}")
    public String dateOfBirth;
    @FixedField(length = 10,format = "\\d{10}")
    public String contactNumber;
    @FixedField(length = 30,format = ".+@.+\\..+")
    public String email;
    @FixedField(length = 30)
    public String residentialAddress;
}
